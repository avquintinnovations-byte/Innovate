uint8_t fontColumn(char c, uint8_t col) {
  if (c < 32 || c > 127) c = ' ';
  uint16_t index = (c - 32) * 5 + col;
  return pgm_read_byte(&font5x7[index]);
}

// ---------- LCD helpers ----------
void lcdWriteCommand(uint8_t cmd){ digitalWrite(LCD_DC,LOW);digitalWrite(LCD_CS,LOW);SPI.transfer(cmd);digitalWrite(LCD_CS,HIGH);}
void lcdWriteData(uint8_t d){ digitalWrite(LCD_DC,HIGH);digitalWrite(LCD_CS,LOW);SPI.transfer(d);digitalWrite(LCD_CS,HIGH);}
void lcdWriteData16(uint16_t d){ lcdWriteData(d>>8); lcdWriteData(d&0xFF);}

void lcdSetAddressWindow(uint16_t x0,uint16_t y0,uint16_t x1,uint16_t y1){
  lcdWriteCommand(0x2A); lcdWriteData16(x0); lcdWriteData16(x1);
  lcdWriteCommand(0x2B); lcdWriteData16(y0); lcdWriteData16(y1);
  lcdWriteCommand(0x2C);
}

void lcdFillScreen(uint16_t color){
  lcdSetAddressWindow(0,0,LCD_W-1,LCD_H-1);
  digitalWrite(LCD_DC,HIGH);digitalWrite(LCD_CS,LOW);
  for(uint32_t i=0;i<(uint32_t)LCD_W*LCD_H;i++){SPI.transfer(color>>8);SPI.transfer(color&0xFF);}
  digitalWrite(LCD_CS,HIGH);
}

void lcdFillRect(uint16_t x,uint16_t y,uint16_t w,uint16_t h,uint16_t color){
  if(x>=LCD_W||y>=LCD_H)return;
  if(x+w>LCD_W)w=LCD_W-x; if(y+h>LCD_H)h=LCD_H-y;
  lcdSetAddressWindow(x,y,x+w-1,y+h-1);
  digitalWrite(LCD_DC,HIGH);digitalWrite(LCD_CS,LOW);
  for(uint32_t i=0;i<(uint32_t)w*h;i++){SPI.transfer(color>>8);SPI.transfer(color&0xFF);}
  digitalWrite(LCD_CS,HIGH);
}

// ---------- text ----------
void lcdDrawChar(uint16_t x,uint16_t y,char c,uint16_t color,uint16_t bg){
  const uint8_t scale=2;
  for(uint8_t col=0;col<5;col++){
    uint8_t bits=fontColumn(c,col);
    for(uint8_t row=0;row<7;row++){
      uint16_t pix=(bits&0x01)?color:bg;
      lcdFillRect(x+col*scale,y+row*scale,scale,scale,pix);
      bits>>=1;
    }
  }
  lcdFillRect(x+5*scale,y,scale,7*scale,bg);
}

void lcdDrawString(uint16_t x,uint16_t y,const String &s,uint16_t color,uint16_t bg){
  for(uint16_t i=0;i<s.length();i++){
    lcdDrawChar(x,y,s[i],color,bg);
    x+=12;
    if(x+12>LCD_W){x=0;y+=16;}
    if(y+16>LCD_H)break;
  }
}
 
bool backlightOn =false ;
bool lastState =HIGH;

// ---------- init ----------
void lcdInit(){
  digitalWrite(LCD_RST,LOW);delay(20);digitalWrite(LCD_RST,HIGH);delay(120);
  lcdWriteCommand(0x11);delay(120);
  lcdWriteCommand(0x3A);lcdWriteData(0x55);      // 16â€‘bit RGB565 [page:23]
  lcdWriteCommand(0x36);lcdWriteData(0x00);      // RGB order [page:23]
  lcdWriteCommand(0x21);lcdWriteCommand(0x13);
  lcdWriteCommand(0x29);delay(20);
}

// ---------- main ----------
void setup(){
  pinMode(SW,INPUT_PULLUP);
  Serial.begin(115200);
  pinMode(LCD_CS,OUTPUT);pinMode(LCD_DC,OUTPUT);
  pinMode(LCD_RST,OUTPUT);pinMode(LCD_BL,OUTPUT);
  digitalWrite(LCD_CS,HIGH);digitalWrite(LCD_BL,HIGH);

  SPI.begin(LCD_SCLK,-1,LCD_MOSI,LCD_CS);
  SPI.beginTransaction(SPISettings(40*1000000,MSBFIRST,SPI_MODE0)); // [page:23]

  lcdInit();
  lcdFillScreen(RED);


}

void loop(){
       
    int s = digitalRead(SW);

  // detect press (HIGH -> LOW edge)
  if (lastState == HIGH && s == LOW) {
    // toggle state
    backlightOn = !backlightOn;

    if (backlightOn) {
      digitalWrite(LCD_BL, HIGH);   // LCD ON
       // <<< USER INPUT IN CODE HERE >>>
  String userText = "AVQUINT";   // change this line to any text

  uint16_t textWidth = 12 * userText.length();
  if(textWidth>LCD_W) textWidth = LCD_W;
  uint16_t x = (LCD_W - textWidth)/2;
  uint16_t y = (LCD_H - 16)/2;

  lcdDrawString(x,y,userText,WHITE,RED);

    } else {
      digitalWrite(LCD_BL, LOW);    // LCD OFF
    }

    delay(100);  // simple debounce
  }

  lastState = s;
  
  }
